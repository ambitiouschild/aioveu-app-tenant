我想把manifest.json里的name复用，更改这个name，小程序对应地方的名字都会更改

在 UniApp 中，`manifest.json`中的 `name`字段是应用的名称，要实现全局修改名称，可以通过以下几种方式：

## 方案1：使用全局变量（推荐）

### 1. 创建全局配置文件

**`common/config.js`**：

```
// 应用配置
export const APP_CONFIG = {
  // 应用名称 - 与 manifest.json 保持一致
  APP_NAME: '我的小程序',
  
  // 版本号
  VERSION: '1.0.0',
  
  // 公司名称
  COMPANY: '我的公司',
  
  // 客服电话
  SERVICE_PHONE: '400-123-4567',
  
  // 其他配置...
  API_BASE_URL: 'https://api.example.com',
  
  // 获取完整应用标题
  getFullTitle(subTitle = '') {
    return subTitle ? `${this.APP_NAME} - ${subTitle}` : this.APP_NAME;
  }
};
```

### 2. 修改页面使用

**`pages/index/index.vue`**：

```
<script setup>
import { APP_CONFIG } from '@/common/config';

// 在页面中使用
onLoad(() => {
  // 设置页面标题
  uni.setNavigationBarTitle({
    title: APP_CONFIG.getFullTitle('首页')
  });
  
  // 或者在模板中使用
  console.log('应用名称:', APP_CONFIG.APP_NAME);
});
</script>

<template>
  <view>
    <!-- 使用应用名称 -->
    <view class="welcome">欢迎来到{{ APP_CONFIG.APP_NAME }}</view>
    <view class="company">{{ APP_CONFIG.COMPANY }} 版权所有</view>
  </view>
</template>
```

## 方案2：通过 Vuex/Pinia 状态管理

### 1. 创建 store

**`store/app.js`**：

```
import { defineStore } from 'pinia';

export const useAppStore = defineStore('app', {
  state: () => ({
    // 从 manifest.json 读取或默认值
    appName: '我的小程序',
    version: '1.0.0',
    company: '我的公司',
    // ... 其他配置
  }),
  
  actions: {
    // 更新应用配置
    updateConfig(config) {
      Object.assign(this, config);
    },
    
    // 获取完整标题
    getFullTitle(subTitle = '') {
      return subTitle ? `${this.appName} - ${subTitle}` : this.appName;
    }
  },
  
  // 持久化存储
  persist: {
    enabled: true,
    strategies: [
      {
        key: 'app-config',
        storage: localStorage
      }
    ]
  }
});
```

### 2. 在组件中使用

```
<script setup>
import { useAppStore } from '@/store/app';

const appStore = useAppStore();

onLoad(() => {
  // 设置页面标题
  uni.setNavigationBarTitle({
    title: appStore.getFullTitle('个人中心')
  });
});
</script>

<template>
  <view>
    <view>{{ appStore.appName }}</view>
    <view>版本: {{ appStore.version }}</view>
  </view>
</template>
```

## 方案3：动态读取 manifest.json（仅H5）

**注意**：小程序环境无法直接读取 `manifest.json`，但可以通过构建脚本处理。

### 1. 创建构建脚本

**`scripts/inject-manifest.js`**：

```
const fs = require('fs');
const path = require('path');

// 读取 manifest.json
const manifestPath = path.resolve(__dirname, '../src/manifest.json');
const manifest = JSON.parse(fs.readFileSync(manifestPath, 'utf8'));

// 生成配置文件
const configContent = `
// 自动生成的配置文件
export const APP_CONFIG = {
  APP_NAME: '${manifest.name || ''}',
  VERSION: '${manifest.versionName || manifest.versionCode || ''}',
  // 其他字段...
  
  // 生成时间
  GENERATED_AT: '${new Date().toISOString()}'
};
`;

// 写入文件
const outputPath = path.resolve(__dirname, '../src/common/app-config.generated.js');
fs.writeFileSync(outputPath, configContent);

console.log('✅ 配置文件生成成功:', outputPath);
```

**`package.json`**：

```
{
  "scripts": {
    "build": "node scripts/inject-manifest.js && uni-build",
    "dev": "node scripts/inject-manifest.js && uni-dev"
  }
}
```

## 方案4：通过环境变量

### 1. 创建环境文件

**`.env`**：

```
# 应用配置
VITE_APP_NAME=我的小程序
VITE_APP_VERSION=1.0.0
VITE_COMPANY_NAME=我的公司
```

**`.env.production`**：

```
VITE_APP_NAME=生产环境名称
VITE_APP_VERSION=1.0.1
VITE_COMPANY_NAME=生产公司
```

### 2. 创建配置工具

**`common/config.js`**：

```
// 从环境变量读取
const config = {
  // 应用名称
  APP_NAME: import.meta.env.VITE_APP_NAME || '默认名称',
  
  // 版本
  VERSION: import.meta.env.VITE_APP_VERSION || '1.0.0',
  
  // 公司
  COMPANY: import.meta.env.VITE_COMPANY_NAME || '默认公司',
  
  // 是否是开发环境
  IS_DEV: import.meta.env.DEV,
  
  // 获取完整标题
  getFullTitle(subTitle = '') {
    const prefix = this.IS_DEV ? '[DEV] ' : '';
    return subTitle ? `${prefix}${this.APP_NAME} - ${subTitle}` : `${prefix}${this.APP_NAME}`;
  }
};

export default config;
```

## 方案5：统一标题管理组件

### 1. 创建标题管理组件

**`components/AppTitle.vue`**：

```
<script setup>
import { ref, computed } from 'vue';
import { APP_CONFIG } from '@/common/config';

const props = defineProps({
  // 子标题
  subTitle: {
    type: String,
    default: ''
  },
  // 是否显示完整标题
  full: {
    type: Boolean,
    default: true
  },
  // 自定义前缀
  prefix: {
    type: String,
    default: ''
  }
});

// 计算完整标题
const fullTitle = computed(() => {
  let title = APP_CONFIG.APP_NAME;
  if (props.prefix) {
    title = `${props.prefix}${title}`;
  }
  if (props.subTitle) {
    title = `${title} - ${props.subTitle}`;
  }
  return title;
});

// 设置导航栏标题
const setNavTitle = (title) => {
  uni.setNavigationBarTitle({
    title: title || fullTitle.value
  });
};

// 暴露方法
defineExpose({
  setNavTitle,
  fullTitle
});
</script>

<template>
  <!-- 可以在任意地方显示应用名称 -->
  <slot v-if="$slots.default" />
  <template v-else>
    <view v-if="full" class="app-title">{{ fullTitle }}</view>
    <view v-else class="app-name">{{ APP_CONFIG.APP_NAME }}</view>
  </template>
</template>

<style scoped>
.app-title {
  font-size: 16px;
  font-weight: bold;
  color: #333;
}

.app-name {
  font-size: 14px;
  color: #666;
}
</style>
```

### 2. 在页面中使用

```
<template>
  <view>
    <!-- 使用组件 -->
    <AppTitle sub-title="首页" />
    
    <!-- 或者 -->
    <AppTitle>
      <view>自定义内容</view>
    </AppTitle>
  </view>
</template>

<script setup>
import AppTitle from '@/components/AppTitle.vue';
import { onReady } from '@dcloudio/uni-app';

onReady(() => {
  // 通过ref调用
  appTitleRef.value.setNavTitle('自定义标题');
});
</script>
```

## 方案6：自动更新所有页面的标题

### 1. 创建全局混入/插件

**`main.js`**：

```
import { createSSRApp } from 'vue';
import App from './App.vue';
import { APP_CONFIG } from './common/config';

export function createApp() {
  const app = createSSRApp(App);
  
  // 全局混入
  app.mixin({
    onLoad() {
      // 自动设置页面标题
      if (this.$page?.meta?.navigationBar?.titleText) {
        const pageTitle = this.$page.meta.navigationBar.titleText;
        const fullTitle = APP_CONFIG.getFullTitle(pageTitle);
        
        uni.setNavigationBarTitle({
          title: fullTitle
        });
      }
    }
  });
  
  // 全局属性
  app.config.globalProperties.$app = APP_CONFIG;
  
  return {
    app
  };
}
```

### 2. 在页面中自动生效

```
<script>
export default {

  //   }

}
```

## 最佳实践组合方案

### 1. 项目结构

```
src/
├── common/
│   ├── config.js          # 配置文件
│   └── constants.js       # 常量
├── store/
│   └── app.js            # 状态管理
├── components/
│   └── AppTitle.vue      # 标题组件
├── utils/
│   └── title.js          # 标题工具
└── main.js               # 全局混入
```

### 2. 完整配置文件

**`common/config.js`**：

```
/**
 * 应用全局配置
 * 修改这里的值，全站生效
 */

// 基础配置
const BASE_CONFIG = {
  // 应用名称 - 修改这里，全站生效
  APP_NAME: '我的商城小程序',
  
  // 版本
  VERSION: '1.0.0',
  
  // 公司
  COMPANY: '科技有限公司',
  
  // 客服
  SERVICE: {
    PHONE: '400-123-4567',
    EMAIL: 'service@example.com',
    WECHAT: 'example_service'
  },
  
  // 版权信息
  COPYRIGHT: `© ${new Date().getFullYear()} 科技有限公司 版权所有`
};

// 运行时配置（可动态修改）
let RUNTIME_CONFIG = { ...BASE_CONFIG };

// 配置管理器
export const AppConfig = {
  // 获取配置
  get(key) {
    return RUNTIME_CONFIG[key];
  },
  
  // 设置配置
  set(key, value) {
    RUNTIME_CONFIG[key] = value;
    this.emitChange(key, value);
  },
  
  // 批量设置
  setAll(config) {
    Object.assign(RUNTIME_CONFIG, config);
    this.emitChange('*', config);
  },
  
  // 重置
  reset() {
    RUNTIME_CONFIG = { ...BASE_CONFIG };
    this.emitChange('reset', RUNTIME_CONFIG);
  },
  
  // 获取完整标题
  getFullTitle(subTitle = '') {
    const title = RUNTIME_CONFIG.APP_NAME;
    return subTitle ? `${title} - ${subTitle}` : title;
  },
  
  // 获取页面标题（用于导航栏）
  getPageTitle(pageTitle = '') {
    return this.getFullTitle(pageTitle);
  },
  
  // 获取关于信息
  getAboutInfo() {
    return {
      name: RUNTIME_CONFIG.APP_NAME,
      version: RUNTIME_CONFIG.VERSION,
      company: RUNTIME_CONFIG.COMPANY,
      copyright: RUNTIME_CONFIG.COPYRIGHT
    };
  },
  
  // 事件监听
  _listeners: new Map(),
  on(event, callback) {
    if (!this._listeners.has(event)) {
      this._listeners.set(event, []);
    }
    this._listeners.get(event).push(callback);
  },
  emitChange(key, value) {
    const listeners = this._listeners.get('change') || [];
    listeners.forEach(cb => cb(key, value));
  }
};

// 默认导出
export default AppConfig;
```

### 3. 使用示例

```
// 在任何地方使用
import AppConfig from '@/common/config';

// 获取应用名称
console.log('应用名称:', AppConfig.get('APP_NAME'));

// 获取完整标题
const pageTitle = AppConfig.getPageTitle('商品详情');

// 动态修改
AppConfig.set('APP_NAME', '新名称');

// 监听配置变化
AppConfig.on('change', (key, value) => {
  console.log(`配置 ${key} 变为:`, value);
});
```

## 总结

**推荐使用方案1 + 方案5的组合**：

1. **`common/config.js`** 统一管理配置
2. **`components/AppTitle.vue`** 组件化显示
3. 关键页面手动设置标题

这样只需要修改 `config.js`中的 `APP_NAME`，所有使用这个配置的地方都会自动更新。
