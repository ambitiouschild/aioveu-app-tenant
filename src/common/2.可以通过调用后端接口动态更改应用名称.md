### 2. 前端配置管理器

**`common/appConfig.js`**：

```
import { getAppConfig, updateAppConfig } from '@/api/app';

// 默认配置（fallback）
const DEFAULT_CONFIG = {
  appName: '我的小程序',
  version: '1.0.0',
  company: '我的公司',
  servicePhone: '400-123-4567',
  themeColor: '#007AFF',
  logo: '/static/logo.png',
  copyright: `© ${new Date().getFullYear()} 我的公司 版权所有`
};

// 本地存储键
const STORAGE_KEYS = {
  CONFIG: 'app_config',
  TIMESTAMP: 'app_config_timestamp',
  VERSION: 'app_config_version'
};

// 缓存时间（1小时）
const CACHE_DURATION = 60 * 60 * 1000;

class AppConfigManager {
  constructor() {
    this.config = { ...DEFAULT_CONFIG };
    this.listeners = new Set();
    this.isLoading = false;
    this.init();
  }

  // 初始化
  async init() {
    // 1. 从本地存储加载
    await this.loadFromStorage();
    
    // 2. 检查是否需要从服务器更新
    if (this.shouldUpdateFromServer()) {
      this.updateFromServer();
    }
    
    // 3. 定时更新（每小时一次）
    this.startAutoUpdate();
  }

  // 从本地存储加载
  async loadFromStorage() {
    try {
      const stored = uni.getStorageSync(STORAGE_KEYS.CONFIG);
      if (stored) {
        this.config = { ...DEFAULT_CONFIG, ...JSON.parse(stored) };
        this.notifyListeners();
      }
    } catch (error) {
      console.warn('加载本地配置失败:', error);
    }
  }

  // 保存到本地存储
  saveToStorage() {
    try {
      uni.setStorageSync(STORAGE_KEYS.CONFIG, JSON.stringify(this.config));
      uni.setStorageSync(STORAGE_KEYS.TIMESTAMP, Date.now());
      uni.setStorageSync(STORAGE_KEYS.VERSION, this.config.version || '1.0.0');
    } catch (error) {
      console.warn('保存配置失败:', error);
    }
  }

  // 检查是否需要从服务器更新
  shouldUpdateFromServer() {
    try {
      const lastUpdate = uni.getStorageSync(STORAGE_KEYS.TIMESTAMP) || 0;
      const now = Date.now();
      
      // 超过缓存时间或从未更新过
      if (now - lastUpdate > CACHE_DURATION) {
        return true;
      }
      
      // 版本号变化
      const storedVersion = uni.getStorageSync(STORAGE_KEYS.VERSION);
      if (storedVersion !== this.config.version) {
        return true;
      }
      
      return false;
    } catch (error) {
      return true; // 出错时强制更新
    }
  }

  // 从服务器更新配置
  async updateFromServer(force = false) {
    if (this.isLoading && !force) return;
    
    this.isLoading = true;
    try {
      const response = await getAppConfig();
      if (response?.data) {
        // 合并配置，服务器数据优先
        this.config = { ...DEFAULT_CONFIG, ...response.data };
        
        // 保存到本地存储
        this.saveToStorage();
        
        // 通知监听者
        this.notifyListeners();
        
        console.log('应用配置已从服务器更新:', this.config);
        return true;
      }
    } catch (error) {
      console.warn('从服务器更新配置失败:', error);
    } finally {
      this.isLoading = false;
    }
    return false;
  }

  // 手动更新配置
  async updateConfig(newConfig) {
    try {
      const response = await updateAppConfig(newConfig);
      if (response?.success) {
        // 更新本地配置
        this.config = { ...this.config, ...newConfig };
        this.saveToStorage();
        this.notifyListeners();
        return true;
      }
    } catch (error) {
      console.error('更新配置失败:', error);
    }
    return false;
  }

  // 获取配置
  get(key) {
    return this.config[key];
  }

  // 获取完整标题
  getFullTitle(subTitle = '') {
    const title = this.config.appName || DEFAULT_CONFIG.appName;
    return subTitle ? `${title} - ${subTitle}` : title;
  }

  // 设置页面标题
  setPageTitle(pageTitle = '') {
    const title = this.getFullTitle(pageTitle);
    uni.setNavigationBarTitle({ title });
  }

  // 监听配置变化
  addListener(callback) {
    this.listeners.add(callback);
    return () => this.listeners.delete(callback);
  }

  // 通知监听者
  notifyListeners() {
    this.listeners.forEach(callback => callback(this.config));
  }

  // 启动定时更新
  startAutoUpdate() {
    setInterval(() => {
      this.updateFromServer();
    }, CACHE_DURATION);
  }

  // 获取所有配置
  getAll() {
    return { ...this.config };
  }
}

// 创建单例
export const appConfig = new AppConfigManager();
export default appConfig;
```

### 5. 全局混入（自动设置页面标题）

**`main.js`**：

import { createSSRApp } from 'vue';
import App from './App.vue';
import { appConfig } from './common/appConfig';

export function createApp() {
  const app = createSSRApp(App);
  
  // 全局混入
  app.mixin({
    onLoad() {
      // 自动设置页面标题
      if (this.$page?.meta?.navigationBar?.titleText) {
        appConfig.setPageTitle(this.$page.meta.navigationBar.titleText);
      }
    }
  });
  
  // 全局属性
  app.config.globalProperties.$appConfig = appConfig;
  
  return { app };
}

## 方案2：WebSocket 实时更新

### 1. WebSocket 服务端推送

```
// 后端 WebSocket
const WebSocket = require('ws');
const wss = new WebSocket.Server({ port: 8080 });

// 配置更新时广播
function broadcastConfigUpdate(newConfig) {
  wss.clients.forEach(client => {
    if (client.readyState === WebSocket.OPEN) {
      client.send(JSON.stringify({
        type: 'CONFIG_UPDATE',
        data: newConfig
      }));
    }
  });
}
```

### 2. 前端 WebSocket 监听

```
// appConfig.js 中添加
class AppConfigManager {
  // ... 原有代码 ...
  
  // 连接 WebSocket
  connectWebSocket() {
    if (this.ws) return;
    
    this.ws = new WebSocket('wss://your-domain.com/ws');
    
    this.ws.onmessage = (event) => {
      const message = JSON.parse(event.data);
      if (message.type === 'CONFIG_UPDATE') {
        this.config = { ...this.config, ...message.data };
        this.saveToStorage();
        this.notifyListeners();
      }
    };
    
    this.ws.onclose = () => {
      setTimeout(() => this.connectWebSocket(), 5000);
    };
  }
}
```

## 优势总结

1. **实时性**：后端修改，前端立即生效
2. **灵活性**：不同环境不同配置
3. **A/B测试**：可对不同用户展示不同配置
4. **热更新**：无需发版即可修改应用信息
5. **版本控制**：可回滚到历史版本
6. **多端同步**：一次修改，多端生效

## 部署建议

1. **配置管理后台**：给运营人员使用
2. **配置版本控制**：记录每次修改
3. **灰度发布**：先对部分用户生效
4. **监控告警**：配置异常时告警
5. **本地缓存**：网络异常时使用缓存
6. **CDN加速**：配置资源放CDN

这样你就可以通过后端接口随时更改应用名称和其他配置，前端会自动更新。
