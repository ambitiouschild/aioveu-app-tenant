### ✅ 3️⃣ handlePay（调用支付）

```js
const handlePay = async () => {
  if (paying.value) return;
  paying.value = true;

  uni.showLoading({ title: "支付中...", mask: true });

  try {
    const amountInFen = Math.round(paymentAmount.value * 100);

    // ✅ 这里用 paymentNo
    const response = await pay({
      orderSn: orderSn.value,
      paymentNo: paymentNo.value,
      paymentMethod: payType.value,
      paymentAmount: amountInFen,
    });

    if (response.code !== "00000") {
      throw new Error(response.message || "创建支付失败");
    }

    const paymentData = response.data;

    let payResult = null;
    if (payType.value === "WX_JSAPI") {
      payResult = await handleJsapiPay(paymentData);
    } else {
      throw new Error("不支持的支付方式");
    }

    uni.hideLoading();

    if (payResult?.success) {
      uni.showToast({
        title: "支付成功",
        icon: "success",
        duration: 1500,
      });

      // ✅ 跳转成功页：用 orderSn
      setTimeout(() => {
        uni.redirectTo({
          url: `/packageD/pages/money/paySuccess?orderSn=${orderSn.value}&amount=${paymentAmount.value}`,
        });
      }, 1500);
    } else {
      throw new Error("支付结果异常");
    }
  } catch (error) {
    uni.hideLoading();
    console.error("支付失败:", error);
    uni.showModal({
      title: "支付失败",
      content: error.message || "请重试",
      showCancel: false,
    });
  } finally {
    paying.value = false;
  }
};
```

✅ 支付用 `paymentNo`
✅ 跳转用 `orderSn`
✅ 前端再也不会“成功但 null”

## ✅ 4️⃣ handleJsapiPay（只验证，不跳转）

```js
const handleJsapiPay = (paymentData) => {
  return new Promise((resolve, reject) => {
    uni.requestPayment({
      provider: "wxpay",
      ...JSON.parse(paymentData.payParams),
      success: async () => {
        try {
          const res = await PayAPI.verifyPaymentStatus(paymentData.paymentNo);
          if (res.paymentStatus === 2) {
            resolve({ success: true });
          } else {
            reject(new Error("支付状态验证失败"));
          }
        } catch (e) {
          reject(e);
        }
      },
      fail: (err) => {
        reject(err);
      },
    });
  });
};
```
