## 方案1：创建全局分享配置（最简单）

**步骤1：创建全局分享配置文件**

```
// config/share-config.js
// 全局分享配置
export const globalShareConfig = {
  title: "可我不敌心软~",  // 统一标题

// 获取页面特定的分享配置
export function getPageShareConfig(pageName, customConfig = {}) {
  // 不同页面的配置
  const pageConfigs = {
    'pages/index/index': {
      title: '首页分享',
      path: '/pages/index/index'
    },

    'pages/user/center': {

  return {
    ...globalShareConfig,
    ...pageConfig,
    ...customConfig
  }
}
```

**步骤2：创建公共分享函数**

```
// utils/share-helper.js
import { getPageShareConfig } from '@/config/share-config'

/**
 * 统一的分享函数
 * @param {string} pageName 页面名称
 * @param {object} customConfig 自定义配置
 */
export function setupShare(pageName, customConfig = {}) {
  const shareConfig = getPageShareConfig(pageName, customConfig)
  
  // 分享给朋友
  uni.onShareAppMessage(() => ({
    title: shareConfig.title,
    path: shareConfig.path + (shareConfig.query ? `?${shareConfig.query}` : ''),
    imageUrl: shareConfig.imageUrl || shareConfig.defaultImageUrl,
    success: (res) => {
      console.log("分享成功", res)
    },
    fail: (err) => {
      console.log("分享失败", err)
    }
  }))

  // 分享到朋友圈
  uni.onShareTimeline(() => ({
    title: shareConfig.title,
    query: shareConfig.query || shareConfig.defaultQuery,
    imageUrl: shareConfig.imageUrl || shareConfig.defaultImageUrl,
    success: (res) => {
      console.log("分享到朋友圈成功", res)
    },
    fail: (err) => {
      console.log("分享到朋友圈失败", err)
    }
  }))
}
```

## 方案3：通过全局事件总线

**步骤1：创建全局事件管理器**

```
// utils/share-manager.js
class ShareManager {
  constructor() {
    this.config = {
      title: "可我不敌心软~",
      defaultImage: "https://cdn.aioveu.com/aioveu-server/avatar/avatar.png",
      defaultQuery: "key=value"
    }
    
    this.listeners = []
  }
  
  // 更新全局配置
  updateGlobalConfig(newConfig) {
    this.config = { ...this.config, ...newConfig }
    this.notifyListeners()
  }
  
  // 注册监听器
  register(callback) {
    this.listeners.push(callback)
  }
  
  // 通知所有监听器
  notifyListeners() {
    this.listeners.forEach(callback => callback(this.config))
  }
  
  // 获取当前配置
  getConfig() {
    return { ...this.config }
  }
}

// 创建单例
export const shareManager = new ShareManager()
```

## 方案5：通过构建脚本自动替换

**如果需要批量修改现有代码，可以使用构建脚本：**

```
// scripts/replace-share.js
const fs = require('fs')
const path = require('path')

// 统一的分享配置
const SHARE_CONFIG = {
  title: '"可我不敌心软~"',
  imageUrl: '"https://cdn.aioveu.com/aioveu-server/avatar/avatar.png"',
  query: '"key=value"'
}

// 遍历所有页面文件
function replaceShareConfig(dir) {
  const files = fs.readdirSync(dir)
  
  files.forEach(file => {
    const filePath = path.join(dir, file)
    const stat = fs.statSync(filePath)
    
    if (stat.isDirectory()) {
      replaceShareConfig(filePath)
    } else if (file.endsWith('.vue') || file.endsWith('.js')) {
      let content = fs.readFileSync(filePath, 'utf-8')
      
      // 替换分享配置
      content = content.replace(
        /title:\s*["'][^"']*["']/g,
        `title: ${SHARE_CONFIG.title}`
      )
      
      content = content.replace(
        /imageUrl:\s*["'][^"']*["']/g,
        `imageUrl: ${SHARE_CONFIG.imageUrl}`
      )
      
      content = content.replace(
        /query:\s*["'][^"']*["']/g,
        `query: ${SHARE_CONFIG.query}`
      )
      
      fs.writeFileSync(filePath, content, 'utf-8')
      console.log(`已更新: ${filePath}`)
    }
  })
}

// 执行替换
replaceShareConfig('./src/pages')
```

## 推荐方案

### 对于新项目

## 总结

要统一修改分享内容，您可以根据项目情况选择：

1. **创建全局配置对象**（最简单）
2. **使用 mixin 或基类**（面向对象）
3. **通过事件总线**（响应式更新）
4. **批量替换脚本**（适用于已有项目）

无论选择哪种方案，关键是将分享配置提取到统一的地方，避免在多个页面中重复定义，这样修改时只需改一处即可。
